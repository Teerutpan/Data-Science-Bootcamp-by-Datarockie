Diamonds Relation between Carat and Price divided by Quality of Cut

ggplot(small_df, aes(carat, price, col=cut)) +
  geom_point(alpha = 0.5, size = 2) +
  geom_smooth() +
  facet_wrap(~cut, nrow = 5) +
  labs(title = "diamonds relation between carat and price divided by quality of cut",
       caption = "source : diamonds at ggplot2") +
  theme_minimal()
